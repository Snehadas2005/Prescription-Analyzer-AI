# INSTALLATION NOTES
# ============================================================================
#
# Quick Install:
#   pip install -r requirements.txt
#
# For GPU Support (if you have CUDA):
#   pip uninstall torch torchvision
#   pip install torch torchvision --index-url https://download.pytorch.org/whl/cu118
#
# Download TrOCR Model (one-time, ~300MB):
#   python -c "from transformers import TrOCRProcessor, VisionEncoderDecoderModel; \
#              TrOCRProcessor.from_pretrained('microsoft/trocr-base-handwritten'); \
#              VisionEncoderDecoderModel.from_pretrained('microsoft/trocr-base-handwritten')"
#
# Alternative TrOCR Models:
#   - microsoft/trocr-small-handwritten (faster, smaller, less accurate)
#   - microsoft/trocr-large-handwritten (slower, larger, more accurate)
#   - microsoft/trocr-base-printed (for printed text only)
#
# Memory Requirements:
#   - Minimum: 4GB RAM (with quantization)
#   - Recommended: 8GB RAM
#   - GPU: Optional (speeds up inference 3-5x)
#
# Disk Space:
#   - Base requirements: ~2GB
#   - TrOCR model: ~300MB
#   - Cache: ~500MB
#   - Total: ~3GB
#
